#include <iostream>
using namespace std;
int main()
{
int x,i,n,a[100],s,d,m,gasit=0,k=0;
cin>>n;
for(i=1;i<=n;i++) cin>>a[i];
cin>>x;
s=1; d=n;
while(s<=d&&!gasit)
{
m=(s+d)/2;
if(a[m]==x) gasit=1;
else if(a[m]<x) s=m+1;
else d=m-1;
}
if(gasit) cout<<m;
else cout<<"Nu se gaseste";

}
