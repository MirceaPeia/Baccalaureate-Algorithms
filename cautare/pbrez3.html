#include <iostream>
int n, a[20],k;
void citire()
{ int i;
cin>>n;
for(i=1;i<=n;i++) cin>>a[i];
cin>>k;
}

void cautbin(int k)
{ int s,d,gasit,p=0;
gasit=0;
s=1; d=n;
while(!gasit &&s<=d)
{ int m=(s+d)/2;
if(a[m]==k) { gasit=1;
cout<<"gasit pe pozitia "<<m<<" in "<<p<<" pasi";
}
else if(a[m]<k) s=m+1;
else d=m-1;
p++;
}
if(!gasit) cout<<"nu s-a gasit in "<<p<<" pasi";
}

int main()
{ citire();
cautbin(k);
return 0;}
