#include <iostream>
using namespace std;
int main()
{
int n,c,p=1,x=0;
cout<<"n="; cin>>n;
cout<<"c="; cin>>c;
while(n!=0)
{
if(n%10!=c)
{
x=x+n%10*p;
p=p*10;
}
n=n/10;
}
cout<<x;
return 0;
}
