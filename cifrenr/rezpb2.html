#include <iostream> 
using namespace std;
int main()
{
int n,np,ni,pp,pi;
cin>>n;
while(n>0)
{
ni=np=0;
pp=pi=1;
while(n)
{
if(n%2==0) {np=np+n%10*pp;
pp=pp*10;
}
else {ni=ni+n%10*pi;
pi=pi*10;
}
n=n/10;
}
if(np>ni) cout<<np<<" ";
else cout<<ni<<" ";
cin>>n;
} 
return 0; 
}
